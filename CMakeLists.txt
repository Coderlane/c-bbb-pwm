SET(PROJECT_NAME "PulseSignalEmitter")

CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
PROJECT(${PROJECT_NAME})

ENABLE_TESTING()

SET(PSE_LIB "pse")
SET(PSE_INCLUDE "${PROJECT_SOURCE_DIR}/include")
SET(PSE_SRC "${PROJECT_SOURCE_DIR}/src")

# Uncomment to make makefiles verbose
# set(CMAKE_VERBOSE_MAKEFILE ON)

# gcc flags
SET(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -std=c11 -Wall -Wextra")
SET(CMAKE_C_FLAGS_DEBUG "${CMAKE_C_FLAGS_DEBUG} -g -DDEBUG")
SET(CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -DNDEBUG -O3")

# Source Code
ADD_SUBDIRECTORY(${PSE_SRC})
INCLUDE_DIRECTORIES(${PSE_INCLUDE})

# LIBRARY
ADD_LIBRARY(${PSE_LIB} SHARED ${SOURCE_FILES})

# Tests
ADD_SUBDIRECTORY(test)
